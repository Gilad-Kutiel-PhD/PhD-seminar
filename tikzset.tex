\usetikzlibrary{
	positioning
	,plotmarks
	,shapes.callouts	
	,intersections	
}

\tikzset{
	on/.code args={<#1>#2}{
  		\only<#1>{\pgfkeysalso{#2}}
	},
	alt/.code args={<#1>#2#3}{
  		\alt<#1>{\pgfkeysalso{#2}}{\pgfkeysalso{#3}}
	},
}

\tikzset{
	note/.style={
		rectangle callout
		,fill=#1
		,thin
		,inner sep=1mm
	}
	,noteorange/.style={note=orange!50}
}

\tikzset{
	label/.style={
		rectangle,	
		draw=none,
		fill=white,
		font=\tiny,
		inner sep=0,
		minimum height=0,
		minimum width=0,
	},
	edge label/.style={
		label,
		midway,
		sloped,
	},
	label below/.style={edge label, below=1mm},
	label above/.style={edge label, above=1mm},
}

\tikzset{
	hide/.style={
		opacity=0
	},
	show/.style={
		opacity=1
	},
	hide nodes/.style={
		every node/.append style={hide}
	},
	hide nodes text/.style={
		every node/.append style={text=white}
	},
	show nodes/.style={
		every node/.append style={show}
	},
	show nodes text/.style={
		every node/.append style={text=black}
	},
	blur/.style={
		opacity=.2
	}
}

% DEFAULTS
\tikzset{default node/.style={
	draw, 
	circle,
	inner sep=0mm,
	minimum size=5mm,
	very thick,
	font=\small,
	black!70,
}}

\tikzset{
	->
	,>=latex
	,every node/.style={default node}
	,every picture/.append style={very thick, black!70}
}
